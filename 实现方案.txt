# MCP 桌宠完整实现方案

本方案实现一个基于 Electron 透明悬浮窗和 Canvas 精灵图动画的桌面宠物，通过 MCP (Model Context Protocol) SSE 服务器对外暴露操控接口，使任何支持 MCP 的 AI 助手（Claude Desktop、Continue、Cline 等）都能够实时控制宠物的动作、表情与气泡文字，作为 AI 工作状态的“进度指示器”。

> 特点：独立运行，不依赖 Codex 客户端；支持社区下载的 Codex 风格精灵图；拖拽移动，鼠标穿透；标准化 MCP 接口。

---

## 1. 技术栈

| 层面         | 技术选择 |
|--------------|----------|
| 桌面容器     | Electron（透明、无边框、置顶）|
| 动画渲染     | HTML5 Canvas + JavaScript（手动帧动画）|
| 动画资源     | 精灵图 (spritesheet.webp) + 动画配置 (pet.json) |
| 交互协议     | MCP (`@modelcontextprotocol/sdk`，SSE 传输) |
| 后端服务     | Express（提供 MCP 的 `/sse` 和 `/messages` 端点）|

## 5. Electron 主进程 (`main.js`)

```javascript
const { app, BrowserWindow } = require('electron');
const path = require('path');
const { startMcpServer } = require('./mcp-server');

app.whenReady().then(() => {
  // 创建透明悬浮窗
  const petWindow = new BrowserWindow({
    width: 300,
    height: 300,
    transparent: true,          // 透明背景
    frame: false,               // 无边框
    alwaysOnTop: true,          // 置顶
    skipTaskbar: true,          // 任务栏不显示
    resizable: false,
    webPreferences: {
      preload: path.join(__dirname, 'preload.js'),
      contextIsolation: true,
      nodeIntegration: false
    }
  });

  // 允许鼠标穿透（用户可点击宠物后的内容）
  petWindow.setIgnoreMouseEvents(true, { forward: true });

  // 加载渲染页面
  petWindow.loadFile('renderer/index.html');

  // 将窗口对象挂载到全局，供 MCP 服务器使用
  global.petWindow = petWindow;

  // 启动 MCP 服务器
  startMcpServer();
});

// 当所有窗口关闭时退出应用（macOS 除外）
app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') app.quit();
});
```

## 7. MCP 服务器 (`mcp-server.js`)

```javascript
const { Server } = require('@modelcontextprotocol/sdk/server/index.js');
const { SSEServerTransport } = require('@modelcontextprotocol/sdk/server/sse.js');
const express = require('express');

let server;

async function startMcpServer() {
  server = new Server(
    {
      name: 'coding-pet',
      version: '1.0.0'
    },
    {
      capabilities: { tools: {} }
    }
  );

  // 定义可用工具列表
  server.setRequestHandler('tools/list', async () => ({
    tools: [
      {
        name: 'set_pet_state',
        description: '设置桌宠的状态（如思考中、完成、失败等）',
        inputSchema: {
          type: 'object',
          properties: {
            state: {
              type: 'string',
              enum: ['idle', 'thinking', 'executing', 'done', 'error']
            }
          },
          required: ['state']
        }
      },
      {
        name: 'pet_say',
        description: '让桌宠显示气泡文字',
        inputSchema: {
          type: 'object',
          properties: {
            text: { type: 'string' }
          },
          required: ['text']
        }
      }
    ]
  }));

  // 处理工具调用
  server.setRequestHandler('tools/call', async (request) => {
    const { name, arguments: args } = request.params;
    if (global.petWindow) {
      global.petWindow.webContents.send('pet-action', { name, ...args });
    }
    return {
      content: [{ type: 'text', text: `宠物已执行 ${name}` }]
    };
  });

  const app = express();

  // SSE 端点
  app.get('/sse', async (req, res) => {
    const transport = new SSEServerTransport('/messages', res);
    await server.connect(transport);
  });

  // 消息端点（SSE 客户端会向这里 POST 数据）
  app.post('/messages', express.json(), (req, res) => {
    res.status(200).send();
  });

  const PORT = 3099;
  app.listen(PORT, () => {
    console.log(`🐱 MCP 桌宠服务器已启动: http://localhost:${PORT}/sse`);
  });
}

module.exports = { startMcpServer };
```

### 8.2 动画逻辑 (`renderer/pet.js`)

```javascript
// ========== 精灵图动画引擎 ==========
const canvas = document.getElementById('pet-canvas');
const ctx = canvas.getContext('2d');
const bubble = document.getElementById('speech-bubble');

let petConfig = null;
let spritesheet = new Image();
let currentAnimation = null;
let currentFrameIndex = 0;
let frameTimer = 0;
let lastTime = 0;
let animFrameId = null;

// 加载宠物配置
fetch('../assets/pet.json')
  .then(res => res.json())
  .then(config => {
    petConfig = config;
    spritesheet.src = config.spritesheetPath;
    spritesheet.onload = () => {
      startAnimation('idle'); // 默认待机
    };
  })
  .catch(err => console.error('宠物配置加载失败:', err));

// 启动指定动画
function startAnimation(name) {
  if (!petConfig || !petConfig.animations[name]) {
    console.warn(`动画 "${name}" 不存在`);
    return;
  }
  currentAnimation = petConfig.animations[name];
  currentFrameIndex = 0;
  frameTimer = 0;
  lastTime = performance.now();
  if (animFrameId) cancelAnimationFrame(animFrameId);
  tick(lastTime);
}

// 每帧刷新
function tick(now) {
  if (!currentAnimation) return;
  const elapsed = now - lastTime;
  lastTime = now;
  frameTimer += elapsed;

  const frameDuration = currentAnimation.frameDuration || 200;
  if (frameTimer >= frameDuration) {
    frameTimer -= frameDuration;
    currentFrameIndex = (currentFrameIndex + 1) % currentAnimation.frames.length;
  }

  const frame = currentAnimation.frames[currentFrameIndex];
  const fw = petConfig.frameSize.width;
  const fh = petConfig.frameSize.height;
  let sx, sy;

  if (typeof frame === 'number') {
    const cols = Math.floor(spritesheet.naturalWidth / fw);
    sx = (frame % cols) * fw;
    sy = Math.floor(frame / cols) * fh;
  } else {
    // 兼容直接给出坐标的格式
    sx = frame.x;
    sy = frame.y;
  }

  ctx.clearRect(0, 0, canvas.width, canvas.height);
  ctx.drawImage(spritesheet, sx, sy, fw, fh, 0, 0, canvas.width, canvas.height);

  animFrameId = requestAnimationFrame(tick);
}

// ========== MCP 动作响应 ==========
const stateToAnimation = {
  idle: 'idle',
  thinking: 'waiting',    // 思考时可用等待动画
  executing: 'running',   // 执行中可用跑动
  done: 'jumping',        // 完成时跳跃
  error: 'failed'         // 失败时播放失败动画
};

window.petAPI.onAction((action) => {
  if (action.name === 'set_pet_state') {
    const animName = stateToAnimation[action.state] || 'idle';
    startAnimation(animName);
  } else if (action.name === 'pet_say') {
    // 显示气泡文字
    bubble.textContent = action.text;
    bubble.classList.add('show');
    clearTimeout(bubble._timeout);
    bubble._timeout = setTimeout(() => {
      bubble.classList.remove('show');
    }, 4000);
  }
});
```

> 可根据需要调整 `stateToAnimation` 映射，将 AI 状态映射到精灵图中实际存在的动作。

## 9. 运行与集成

### 9.1 启动桌宠

确保精灵图 `spritesheet.webp` 与 `pet.json` 放在 `assets/` 目录下，然后运行：

```bash
npm start
```

桌宠窗口将出现在屏幕左上角，并自动播放待机动画。MCP 服务器同时监听 `http://localhost:3099/sse`。

### 9.2 配置 AI 助手连接

以 **Claude Desktop** 为例，在 `claude_desktop_config.json` 中添加：

```json
{
  "mcpServers": {
    "coding-pet": {
      "url": "http://localhost:3099/sse"
    }
  }
}
```

同样，**Continue** (VS Code 插件) 在 `config.json` 中：

```json
{
  "experimental": {
    "mcpServers": {
      "coding-pet": {
        "url": "http://localhost:3099/sse"
      }
    }
  }
}
```

**Cline** (VS Code) 在其设置面板中添加 MCP 服务器地址即可。

### 9.3 让 AI 自动触发状态变化

在 AI 助手的 System Prompt 中加入以下指令：

```
当您开始处理一个任务时，立即调用 set_pet_state，参数为 {"state": "thinking"}。
当您正在生成代码或执行操作时，调用 {"state": "executing"}。
任务成功完成后，调用 {"state": "done"}，随后恢复到 {"state": "idle"}。
如果遇到错误或无法完成，调用 {"state": "error"}。
适当时可以使用 pet_say 让宠物说出简短的消息。
```

这样，AI 就会在整个处理流程中自动更新宠物状态，桌宠成为真正的“AI 进度条”。

## 10. 进阶功能（可选）

- **拖拽移动**：可在 Electron 主进程中监听 `move` 事件，或通过渲染进程 `-webkit-app-region: drag` 实现。
- **交互模式**：单击宠物时切换鼠标穿透状态，以便拖拽或点击宠物。
- **状态持久化**：将宠物位置等信息保存到本地文件，重启后恢复。
- **更多动画**：替换自己的精灵图，调整 `pet.json` 即可扩展。

## 附录：常见问题

**Q：宠物显示不全或帧偏移？**
A：确保 `pet.json` 中的 `frameSize` 与精灵图实际切割一致，且 `spritesheet.naturalWidth` 能被帧宽整除。

**Q：MCP 连接失败？**
A：检查防火墙是否放行 3099 端口，确认 MCP 服务器已随 Electron 启动。

**Q：如何获取更多宠物？**
A：访问 Codex 宠物社区网站（如 `codexpet.xyz`、`petdex.crafter.run`）下载打包好的精灵图和配置，放入 `assets` 目录并适当修改 `pet.json` 即可。
